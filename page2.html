var R1 = 100,
   R2 = 70,
   R3 = 25,
   Rsarja;
 
 function kolmeSarjaan() {
   Rsarja = R1 + R2 + R3;
   return Rsarja;
 }
 function kolmeRinnan() {
   Rsarja = 1 / (1 / R1 + 1 / R2 + 1 / R3);
   return Rsarja;
 }
 function ex8() {
   document.getElementById(
     "sarjaan"
   ).innerHTML = `Sarjaan (R1 ${R1}, R2 ${R2}, R3 ${R3}): ${kolmeSarjaan(
     R1,
     R2,
     R3
   )}omh`;
   document.getElementById(
     "rinnan"
   ).innerHTML = `Rinnan (R1 ${R1}, R2 ${R2}, R3 ${R3}): ${kolmeRinnan(
     R1,
     R2,
     R3
   )}omh`;
 }
